# Usage

# Text input

Example

<div>
    <label class='form'>Text input</label>
    <input type='text' class='form'>
</div>

# Textarea input

Example

<div>
    <label class="form">Textarea input</label>
    <textarea class="form"></textarea>
</div>

# Select input

Example

<div>
    <label class="form">Select input</label>
    <select class="form">
        <option>Select</option>
        <option>First option</option>
        <option>Second option</option>
    </select>
</div>

# File input

Example

<div>
    <label class="form">File input</label>
    <input type="file" class="form">
</div>

# Checkbox group

Example

<div>
    <label class="form">Checkbox group</label>
</div>
<div>
    <input type="checkbox" class="form">
    <label class="form">First checkbox</label>
</div>
<div>
    <input type="checkbox" class="form" checked>
    <label class="form">Second checkbox</label>
</div>

# Radio group

Example

<div>
    <label class="form">Radio group</label>
</div>
<div>
    <input type="radio" name="radio" class="form">
    <label class="form">First radio</label>
</div>
<div>
    <input type="radio" name="radio" class="form" checked>
    <label class="form">Second radio</label>
</div>

# Button

# Button types

Example

Link button 
<a class="form" href="#">Link button</a>
<button class="form" type="submit">Button</button>
<input class="form" type="button" value="Input button">
<input class="form" type="submit" value="Input submit">

# Button variants

Example

<button type="button" class="form">Primary</button>
<button type="button" class="form success">Success</button>
<button type="button" class="form info">Info</button>
<button type="button" class="form warning">Warning</button>
<button type="button" class="form danger">Danger</button>

# Button full width

Example

<button class="form full" type="button">Button</button>
